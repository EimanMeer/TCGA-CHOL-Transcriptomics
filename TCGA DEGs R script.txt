# 5. PREPARE COLDATA (Metadata for Design)

  sample_id = colnames(countData),

) %>% column_to_rownames("sample_id")

library(DESeq2)
# 6. RUN THE UNBIASED DGE ANALYSIS
cat("Step 2: Running DESeq2 (Negative Binomial Model)...\n")
dds <- DESeqDataSetFromMatrix(countData = countData, colData = colData, design = ~ condition)
dds <- dds[rowSums(counts(dds)) > 10, ] # Filter low expression genes
dds <- DESeq(dds)
res <- results(dds)

# 7. ANNOTATE RESULTS WITH BIOTYPES
res_df <- as.data.frame(res) %>%
  rownames_to_column("gene_id") %>%
  left_join(raw_data %>% dplyr::select(gene_id, gene_name, gene_type), by = "gene_id") %>%
  arrange(padj)

# 8. EXPORT TOP RESULTS
write.csv(res_df, "E:/TCGA_CHOL_Discovery_DGE_Full.csv", row.names = FALSE)

# ==============================================================================
# PHASE 3: CORRELATION ANALYSIS (BUILDING THE AXIS)

# 1. Get Normalized Transformation (VST)
vsd <- vst(dds, blind = FALSE)
vst_matrix <- assay(vsd)

# 2. Pick the candidates from your 'sig_list'
# Let's pick the top 5 lncRNAs and top 5 mRNAs based on significance
top_lnc_ids <- sig_list %>% filter(gene_type == "lncRNA") %>% head(5) %>% pull(gene_id)
top_m_ids   <- sig_list %>% filter(gene_type == "protein_coding") %>% head(5) %>% pull(gene_id)

# 3. Create a small matrix for just these candidates
axis_data <- t(vst_matrix[c(top_lnc_ids, top_m_ids), ])

# 4. Calculate Spearman Correlation
# This tells us which lncRNA and mRNA move together
cor_results <- cor(axis_data, method = "spearman")

# 5. Visualize the Axis with a Heatmap
library(pheatmap)
pheatmap(cor_results, 
         main = "Regulatory Axis Correlation (lncRNA vs mRNA)",
         display_numbers = TRUE,
         color = colorRampPalette(c("blue", "white", "red"))(100))
